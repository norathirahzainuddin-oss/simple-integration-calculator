<script>
    let myChart = null;

    function calculateIntegration() {
        const formula = document.getElementById('function').value;
        const a = parseFloat(document.getElementById('limit-a').value);
        const b = parseFloat(document.getElementById('limit-b').value);
        
        const errorBox = document.getElementById('error-box');
        const resultBox = document.getElementById('result-box');
        const resultValue = document.getElementById('result-value');
        
        // Reset views
        errorBox.style.display = 'none';
        resultBox.style.display = 'none';

        try {
            // Compile the math expression using math.js
            const expr = math.compile(formula);
            
            // 1. Calculate Definite Integral using Simpson's Rule (Numerical Approximation)
            const n = 1000; // number of intervals (even number)
            const h = (b - a) / n;
            let sum = expr.evaluate({x: a}) + expr.evaluate({x: b});
            
            for (let i = 1; i < n; i++) {
                const x = a + i * h;
                if (i % 2 === 0) {
                    sum += 2 * expr.evaluate({x: x});
                } else {
                    sum += 4 * expr.evaluate({x: x});
                }
            }
            const integralResult = (h / 3) * sum;
            
            if (isNaN(integralResult)) {
                throw new Error("Calculation resulted in an undefined value. Check your function limits.");
            }

            // Display Result rounded to 4 decimal places
            resultValue.innerHTML = `∫ f(x) dx = <strong>${integralResult.toFixed(4)}</strong>`;
            resultBox.style.display = 'block';
            
            // 2. Generate Data for Charting
            generateChart(expr, a, b);

        } catch (error) {
            errorBox.innerText = "Error: " + error.message;
            errorBox.style.display = 'block';
        }
    }

    function generateChart(compiledExpr, a, b) {
        // Pad graph slightly wider than limits for aesthetic context
        const padding = Math.abs(b - a) * 0.3 || 1; 
        const startX = a - padding;
        const endX = b + padding;
        
        const labels = [];
        const mainDatasetData = [];
        const integralDatasetData = [];

        const points = 150;
        const step = (endX - startX) / points;

        for (let i = 0; i <= points; i++) {
            const currentX = startX + (i * step);
            let currentY = 0;
            
            try {
                currentY = compiledExpr.evaluate({x: currentX});
            } catch(e) {
                currentY = null;
            }

            labels.push(currentX.toFixed(2));
            mainDatasetData.push(currentY);

            // If x is within the limits of integration [a, b], map it to the shade area
            if (currentX >= a && currentX <= b) {
                integralDatasetData.push(currentY);
            } else {
                integralDatasetData.push(null); // Out of bounds, won't shade
            }
        }

        // Destroy previous chart instance if it exists to prevent overlapping redraws
        if (myChart) {
            myChart.destroy();
        }

        const ctx = document.getElementById('myChart').getContext('2d');
        myChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [
                    {
                        label: 'f(x)',
                        data: mainDatasetData,
                        borderColor: '#2c3e50',
                        borderWidth: 2,
                        pointRadius: 0,
                        fill: false
                    },
                    {
                        label: 'Integrated Area',
                        data: integralDatasetData,
                        backgroundColor: 'rgba(52, 152, 219, 0.3)',
                        borderColor: 'transparent',
                        pointRadius: 0,
                        fill: 'origin', // fills to y=0 line
                    }
                ]
            },
            options: {
                responsive: true,
                scales: {
                    x: {
                        title: { display: true, text: 'X axis' },
                        ticks: { maxTicksLimit: 10 }
                    },
                    y: {
                        title: { display: true, text: 'Y axis' }
                    }
                },
                plugins: {
                    legend: { display: true }
                }
            }
        });
    }

    // Run once on load to show initial example
    window.onload = calculateIntegration;
</script>
